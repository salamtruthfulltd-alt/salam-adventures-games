async function generateWorksheet() { const topic = document.getElementById('topic').value; const btn = document.getElementById('genBtn'); const display = document.getElementById('worksheetDisplay'); if (!topic) return alert("Enter a topic!"); btn.innerText = "Generating..."; display.innerHTML = "Asking the AI..."; document.getElementById('outputArea').classList.remove('hidden'); try { // Using a free public model from Hugging Face (No Key Required for basic use) const response = await fetch( "https://api-inference.huggingface.co/models/HuggingFaceH4/zephyr-7b-beta", { method: "POST", body: JSON.stringify({ inputs: `Create a simple SEN worksheet about ${topic} with 3 steps.` }), } ); const data = await response.json(); display.innerHTML = data[0].generated_text.replace(/\n/g, "
"); } catch (e) { display.innerHTML = "Server busy. Try again in a moment!"; } finally { btn.innerText = "Build Worksheet"; } }
