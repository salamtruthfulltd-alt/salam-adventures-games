async function generateWorksheet() { const topic = document.getElementById('topic').value; const btn = document.getElementById('genBtn'); const display = document.getElementById('worksheetDisplay'); if (!topic) return alert("Please enter a topic first!"); btn.disabled = true; btn.innerText = "⏳ Writing..."; display.innerHTML = "

The AI is crafting your SEN materials...

"; document.getElementById('outputArea').classList.remove('hidden'); try { const response = await fetch( "https://api-inference.huggingface.co/models/MistralAI/Mistral-7B-Instruct-v0.2", { headers: { "Content-Type": "application/json", // OPTIONAL: Add "Authorization": "Bearer hf_xxx" here if you get a free token }, method: "POST", body: JSON.stringify({ inputs: `[INST] Task: Create a SEN worksheet. Topic: ${topic}. Requirements: Use very simple English, 3 clear steps, and 2 multiple choice questions. Format: Clear headings, no fluff. [/INST]`, parameters: { max_new_tokens: 500, temperature: 0.5, // Lower temperature makes it more "literal" and less "creative" (better for SEN) return_full_text: false } }), } ); const data = await response.json(); // 1. Handle Model Loading (Wait-and-retry) if (data.error && data.error.includes("loading")) { display.innerHTML = "🏫 The AI is 'waking up' its brain. Please click Build Worksheet again in 10 seconds!"; return; } // 2. Extract generated text safely let cleanText = ""; if (Array.isArray(data) && data[0].generated_text) { // Some models return the prompt + result, others just the result. // This logic cleans both. cleanText = data[0].generated_text.split('[/INST]').pop().trim(); } else { throw new Error("Unexpected API response"); } // 3. Render to the screen display.innerHTML = cleanText.replace(/\n/g, "
"); } catch (e) { console.error(e); display.innerHTML = "⚠️ The AI server is busy. Give it a moment and try again!"; } finally { btn.disabled = false; btn.innerText = "Build Worksheet"; } }
